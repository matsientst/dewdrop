package com.dewdropper.aggregate;

import com.dewdropper.aggregate.proxy.AggregateProxyFactory;
import com.dewdropper.structure.api.Command;
import com.dewdropper.utils.ReflectionUtils;
import java.util.Map;
import java.util.Optional;
import java.util.Set;
import javassist.util.proxy.ProxyFactory;
import lombok.extern.log4j.Log4j2;
import org.apache.commons.collections4.map.HashedMap;
import org.springframework.stereotype.Component;

@Component
@Log4j2
public class AggregateStateOrchestrator {
    private Map<Class<?>, ProxyFactory> proxyFactories;

    public void init() {
        proxyFactories = new HashedMap<>();
        Set<Class<?>> aggregates = ReflectionUtils.getAnnotatedClasses(Aggregate.class);
        aggregates.forEach(aggregate -> {
            ProxyFactory factory = new ProxyFactory();
            factory.setSuperclass(aggregate);
            proxyFactories.put(aggregate, factory);
        });
    }

    public <T> Optional<T> create(Class<T> clazz) {
        try {
            AggregateProxyFactory factory = new AggregateProxyFactory();

            return Optional.of(factory.create(clazz));
        } catch (ReflectiveOperationException e) {
            log.error("Couldn't create object", e);
            return Optional.empty();
        }
    }

    public void onCommand(Command command) {

    }
}
