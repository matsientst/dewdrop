package com.dewdropper.aggregate;

import static java.util.Objects.requireNonNull;

import com.dewdropper.structure.events.Event;
import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;
import lombok.Data;

@Data
public class EventRecorder implements Serializable {
    private List<Event> recorded;

    public EventRecorder() {
        recorded = new ArrayList<>();
    }

    public boolean hasRecordedEvents() {
        return !recorded.isEmpty();
    }

    public void recordEvent(Event event) {
        requireNonNull(event);

        recorded.add(event);
    }

    public void reset() {
        recorded.clear();
    }


    public List<Event> recordedEvents() {
        return recorded;
    }
}
