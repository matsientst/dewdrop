package com.dewdropper.utils;

import static java.util.stream.Collectors.toSet;
import static org.reflections.scanners.Scanners.MethodsAnnotated;
import static org.reflections.scanners.Scanners.TypesAnnotated;

import java.lang.annotation.Annotation;
import java.lang.reflect.Method;
import java.util.HashSet;
import java.util.Objects;
import java.util.Set;
import org.apache.commons.collections4.CollectionUtils;
import org.reflections.Reflections;
import org.reflections.util.ClasspathHelper;
import org.reflections.util.ConfigurationBuilder;

public class ReflectionUtils {
    private static Reflections methodsReflections = new Reflections(new ConfigurationBuilder().setUrls(ClasspathHelper.forPackage("com.dewdropper")).setScanners(MethodsAnnotated));
    private static Reflections typesAnnotated = new Reflections(new ConfigurationBuilder().setUrls(ClasspathHelper.forPackage("com.dewdropper")).setScanners(TypesAnnotated));

    public static Set<Method> getAnnotatedMethods(Class<? extends Annotation> annotationClass) {
        Set<Method> methods = methodsReflections.getMethodsAnnotatedWith(annotationClass);
        if (CollectionUtils.isNotEmpty(methods)) {
            methods = methods.stream().filter(method -> !Objects.equals(method.getParameterTypes()[0].getSimpleName(), "Object")).collect(toSet());
        }
        return methods;
    }

    public static Set<Class<?>> getAnnotatedClasses(Class<? extends Annotation> annotationClass) {
        Set<Class<?>> classes = typesAnnotated.getTypesAnnotatedWith(annotationClass);
        if (CollectionUtils.isNotEmpty(classes)) { return classes.stream().collect(toSet()); }
        return new HashSet<>();
    }
}
