package com.dewdropper.aggregate;

import static java.util.Objects.requireNonNull;

import com.dewdropper.structure.events.Event;
import java.lang.reflect.InvocationTargetException;
import java.util.ArrayList;
import java.util.List;
import lombok.Data;
import lombok.extern.log4j.Log4j2;
import org.apache.commons.lang3.reflect.MethodUtils;

public class EventStateMachine extends BaseEntity implements AggregateRoot {
    protected EventRecorder recorder;
    private long version;

    public EventStateMachine(EventRecorder recorder, long version) {
        this.recorder = recorder;
        this.version = version;
    }

    public EventStateMachine() {
        this.recorder = new EventRecorder();
        this.version = -1L;
    }


    @Override
    public void restoreFromEvents(List<Event> events) {
        requireNonNull(events);

        if (recorder.hasRecordedEvents()) throw new IllegalStateException("Restoring from events is not possible when an instance has recorded events.");

        events.stream().forEach(event -> {
            if (version < 0) // new aggregates have an expected version of -1 or -2
                version = 0; // got first event (zero based)
            else
                version++;
            applyState(event);
        });
    }

    @Override
    public <T extends Event> void applyState(T event) {
        try {
            MethodUtils.invokeMethod(this, true, "apply", event);
            // log.info("successfully applied " + event.getClass().getSimpleName());
        } catch (InvocationTargetException | IllegalAccessException | NoSuchMethodException e) {
            // log.error("Unable to invoke apply of event: {} - message: {} - for the aggregateRoot",
            // event.getClass().getSimpleName(), e.getMessage(), e);
        }
    }

    @Override
    public long getExpectedVersion() {
        return version;
    }

    @Override
    public void updateWithEvents(List<Event> events, long expectedVersion) {
        requireNonNull(events);

        if (version < 0) throw new IllegalArgumentException("Updating with events is not possible when an instance has no historical events");
        if (version != expectedVersion) { throw new IllegalArgumentException("Expected version mismatch when updating - actual version:" + version + ", expectedVersion:" + expectedVersion); }

        events.stream().forEach(event -> {
            version++;
            applyState(event);
        });
    }

    @Override
    public List<Event> takeEvents() {
        takeEventStarted();
        List<Event> records = new ArrayList<>(recorder.recordedEvents());
        recorder.reset();
        version += records.size();
        takeEventsCompleted();
        return records;
    }

    @Override
    public void takeEventStarted() {
        // Not Implemented yet.
    }

    @Override
    public void takeEventsCompleted() {
        // Not Implemented yet.
    }

    @Override
    public void onEventRaised(Event event) {
        // Not Implemented yet.
    }

    @Override
    public void raise(Event event) {
        onEventRaised(event);
        applyState(event);
        recorder.recordEvent(event);
    }

    public long getVersion() {
        return version;
    }

    public void setVersion(long version) {
        this.version = version;
    }

    @Override
    public boolean equals(Object o) {
        return super.equals(o);
    }

    @Override
    public int hashCode() {
        return super.hashCode();
    }
}
