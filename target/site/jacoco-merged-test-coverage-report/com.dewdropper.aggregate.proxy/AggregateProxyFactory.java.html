package com.dewdropper.aggregate.proxy;

import com.dewdropper.aggregate.EventState;
import java.io.IOException;
import java.lang.reflect.Method;
import java.security.AccessController;
import java.security.PrivilegedAction;
import java.security.ProtectionDomain;
import javassist.CannotCompileException;
import javassist.ClassClassPath;
import javassist.ClassPool;
import javassist.CtClass;
import javassist.CtConstructor;
import javassist.Loader;
import javassist.NotFoundException;
import javassist.util.proxy.MethodHandler;
import javassist.util.proxy.ProxyFactory;
import javassist.util.proxy.ProxyObject;
import lombok.extern.log4j.Log4j2;

@Log4j2
public class AggregateProxyFactory {

    @SuppressWarnings("unchecked")
    public static <T> T create(Class<T> iface) throws ReflectiveOperationException {
        ClassPool pool = ClassPool.getDefault();

        Loader loader = (Loader) AccessController.doPrivileged((PrivilegedAction) () -> new Loader(Thread.currentThread().getContextClassLoader(), pool));
        try {
            CtClass ctClass = pool.get(iface.getName());
            ctClass.defrost();
            ctClass.setSuperclass(pool.get(EventState.class.getName()));
            ctClass.writeFile("com.dewdropper");
            ProxyFactory factory = new ProxyFactory();
            factory.setSuperclass(ctClass.toClass(loader, iface.getProtectionDomain()));

            MethodHandler handler = new AggregateHandler();

            Object instance = factory.create(new Class[] {}, new Object[] {}, handler);
            return (T) instance;
        } catch (ReflectiveOperationException e) {
            log.error("Couldn't create object", e);
            throw e;
        } catch (NotFoundException | CannotCompileException e) {
            log.error("Couldn't create object", e);
            return null;
        } catch (IOException e) {
            throw new RuntimeException(e);
        }
    }
}
